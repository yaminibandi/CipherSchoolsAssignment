#include <bits/stdc++.h> 
using namespace std; 

bool isBalanced(string expr) 
{ 
	stack<char> s; 
	char x; 
	for (int i=0;i<expr.length();i++) { 
		if (expr[i]=='(' || expr[i]=='{' || expr[i]=='[') { 
			s.push(expr[i]); 
			continue; 
		} 
		
		if (s.empty()) 
			return false; 

		if(expr[i]==')')
		{
		    char x=s.top(); 
			s.pop(); 
			if (x!='(') 
				return false; 
		}
		if(expr[i]=='}')
		{
		    char x=s.top(); 
			s.pop(); 
			if (x!='{') 
				return false; 
		}
		if(expr[i]==']')
		{
		    char x=s.top(); 
			s.pop(); 
			if (x!='[') 
				return false; 
		}
	} 
	return s.empty()?true:false; 
} 
 
int main() 
{ 
    int testcases;
    cin>>testcases;
    while(testcases--){
        string expression;
        cin>>expression;
        if (isBalanced(expression)) 
		    cout<<"YES"<<endl; 
	    else
		    cout<<"NO"<<endl; 
    }
	return 0; 
}
